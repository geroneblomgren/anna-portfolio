**Step 1: Add particle CSS to `src/app/globals.css`**

Add the following at the END of the file, AFTER all existing rules. The entire block is inside a `@media (prefers-reduced-motion: no-preference)` guard — particles are completely absent for users who prefer reduced motion.

```css
/* Ambient ink particles — compositor-only CSS animation (PERF-05: zero JS cost) */
@media (prefers-reduced-motion: no-preference) {
  .particle {
    position: fixed;
    width: var(--particle-size, 3px);
    height: var(--particle-size, 3px);
    border-radius: 50%;
    background: var(--color-text-muted, #888);
    pointer-events: none;
    opacity: 0;
    will-change: transform, opacity;
    animation: particle-drift var(--particle-duration, 18s) var(--particle-delay, 0s) infinite ease-in-out;
    z-index: 1;
  }

  @keyframes particle-drift {
    0%   { transform: translate(0, 0) scale(1);          opacity: 0;    }
    15%  { opacity: var(--particle-peak-opacity, 0.12);  }
    70%  { transform: translate(var(--dx, 30px), var(--dy, -60px)) scale(0.6); opacity: 0.06; }
    100% { transform: translate(var(--dx2, -20px), var(--dy2, -120px)) scale(0.3); opacity: 0; }
  }
}
```

Key points:
- **Compositor-only:** Only `transform` and `opacity` are animated — these go directly to the GPU compositor thread and do NOT trigger layout or paint. Unaffected by CPU throttle.
- **`will-change: transform, opacity`:** Promotes each particle to its own compositor layer. With only 6 particles this is safe (under the 10-element ceiling).
- **`pointer-events: none`:** Particles never block interaction (PERF-02).
- **Custom properties per particle:** `--dx`, `--dy`, `--dx2`, `--dy2` control drift direction. `--particle-duration` and `--particle-delay` stagger timing. `--particle-size` varies speck size. `--particle-peak-opacity` controls max visibility.

```tsx
import type { CSSProperties } from 'react'

interface ParticleConfig {
  left: string
  top: string
  dx: string
  dy: string
  dx2: string
  dy2: string
  duration: string
  delay: string
  size: string
  peakOpacity: string
}

const PARTICLES: ParticleConfig[] = [
  { left: '12%', top: '35%', dx: '25px', dy: '-80px', dx2: '-15px', dy2: '-160px', duration: '22s', delay: '0s', size: '3px', peakOpacity: '0.10' },
  { left: '75%', top: '25%', dx: '-30px', dy: '-50px', dx2: '10px', dy2: '-130px', duration: '19s', delay: '4s', size: '2px', peakOpacity: '0.12' },
  { left: '45%', top: '60%', dx: '15px', dy: '-70px', dx2: '-25px', dy2: '-140px', duration: '24s', delay: '8s', size: '4px', peakOpacity: '0.08' },
  { left: '88%', top: '50%', dx: '-20px', dy: '-90px', dx2: '5px', dy2: '-170px', duration: '20s', delay: '2s', size: '2px', peakOpacity: '0.11' },
  { left: '30%', top: '75%', dx: '35px', dy: '-60px', dx2: '-10px', dy2: '-120px', duration: '26s', delay: '12s', size: '3px', peakOpacity: '0.09' },
  { left: '60%', top: '15%', dx: '-15px', dy: '-45px', dx2: '20px', dy2: '-110px', duration: '21s', delay: '6s', size: '2px', peakOpacity: '0.10' },
]

export function AmbientParticles() {
  return (
    <div aria-hidden="true">
      {PARTICLES.map((p, i) => (
        <span
          key={i}
          className="particle"
          style={{
            left: p.left,
            top: p.top,
            '--dx': p.dx,
            '--dy': p.dy,
            '--dx2': p.dx2,
            '--dy2': p.dy2,
            '--particle-duration': p.duration,
            '--particle-delay': p.delay,
            '--particle-size': p.size,
            '--particle-peak-opacity': p.peakOpacity,
          } as CSSProperties}
        />
      ))}
    </div>
  )
}
```

Key design decisions:
1. **No `'use client'` needed** — this component has zero interactivity, no hooks, no event handlers. It renders static HTML with CSS custom properties. It can be a server component (renders once, no JS shipped to client).
2. **6 particles** — conservative count. Each gets its own `will-change: transform` compositor layer. 6 is well under the 10-element safety ceiling. Staggered delays (0s to 12s) mean only 1-2 are visible at any moment.
3. **Varying sizes (2-4px), opacities (0.08-0.12), durations (19-26s)** — creates organic randomness without JS.
4. **All particles drift upward** (negative `--dy` values) — simulates rising smoke/ink specks.
5. **`aria-hidden="true"`** — purely decorative, hidden from assistive technology.
